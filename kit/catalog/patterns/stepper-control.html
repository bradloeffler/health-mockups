Try the buttons — the value clamps between 0 and 10, +/−1 per tap. Buttons disable at the boundary.
<div class="k-row k-row--between">
  <button class="k-step-btn" id="minus" aria-label="Less"><i class="ph ph-minus"></i></button>
  <span class="k-numeric k-numeric--lg">
    <span class="k-numeric__val" id="val">5</span>
    <span class="k-numeric__unit">/ 8</span>
  </span>
  <button class="k-step-btn" id="plus" aria-label="More"><i class="ph ph-plus"></i></button>
</div>

<script>
  const MIN = 0, MAX = 10;
  let n = 5;
  const val = document.querySelector('#val');
  const minus = document.querySelector('#minus');
  const plus  = document.querySelector('#plus');
  function render() {
    val.textContent = n;
    minus.disabled = n <= MIN;
    plus.disabled  = n >= MAX;
  }
  minus.addEventListener('click', () => { n = Math.max(MIN, n - 1); render(); });
  plus .addEventListener('click', () => { n = Math.min(MAX, n + 1); render(); });
  render();
</script>Copy
